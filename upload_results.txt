                function checkHex(n) {

                }

                function Hex2Bin(n) {
                    if (!checkHex(n)) return 0;
                    return parseInt(n, 16).toString(2)
                }

                function Bin2Hex(n) {
                    if (!checkBin(n)) return 0;

                }

                function reverse(s) {
                    return s.split('').reverse().join('');
                }

                function pad(s, z) {
                    s = "" + s;
                    return s.length < z ? pad("0" + s, z) : s
                }

                function isBitSet(hex, bit) {
                    var padded = reverse(pad(Hex2Bin(hex), 32)).split('');
                    return (padded[bit] == '1') ? true : false;
                }

                function flipBit(hex, bit) {
                    var reversePadded = reverse(pad(Hex2Bin(hex), 32)).split('');
                    reversePadded[bit] = (reversePadded[bit] == '1') ? '0' : '1';
                    return Bin2Hex(reverse(reversePadded.join('')));
                }
